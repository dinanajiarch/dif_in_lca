# Define the logits for the first example as a list
logits <- data.frame(
  C1 = c(-2,-2,-2,-2,-2,-2,-2,-2,-2,-2),
  C2 = c(-2,-2,-2,-2,-2,2,2,2,2,2),
  C3 = c(2,2,2,-2,-2,-2,2,2,2,2),
  C4 = c(2,2,2,2,2,-2,-2,-2,-2,-2),
  C5 = c(2,2,2,2,2,2,2,2,2,2)
)


# Convert logits to probabilities using the sigmoid function
sigmoid <- function(x) {
  1 / (1 + exp(x))
}


probabilities <- lapply(logits, sigmoid) %>% 
  as.data.frame() 

df <- probabilities %>% 
  mutate(Indicator = rep(1:10)) %>%
  pivot_longer(cols = -c(Indicator), names_to = "Class", values_to = "Probability")


ggplot(df, aes(x = factor(Indicator), y = Probability, group = Class, color = Class)) +
  geom_line() +
  geom_hline(yintercept = c(0.7, 0.3))+
  geom_point() +
  labs(title = "Ideal Separation (Complex)",
       x = "Indicators", y = "Probabilities") +
  ylim(0,1) +
  theme_cowplot()
